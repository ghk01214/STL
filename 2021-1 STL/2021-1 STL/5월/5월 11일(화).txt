		if (p == m.end())
		{
			std::cout << groupName << "擎 橈朝 斜瑜." << std::endl;
			
			continue;
		}

		std::cout << "斜瑜 " << p->first << "曖 詹幗 : ";

		for (const std::string& s : p->second)
		{
			std::cout << s << " ";
		}

		std::cout << std::endl << std::endl;
	}

	Save("main.cpp");
}
